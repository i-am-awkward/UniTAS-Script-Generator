-- BUILT-IN FUNCTIONS

-- Equips weapon and variation, takes in the key corresponding to your weapon and the key corresponding to your variation bind. Variation key argument is optional
function equipWeapon(weaponKey, variationKey)
    Weapon = weaponKey
    Variation = variationKey or "none"
    key.hold(Weapon)
    movie.frame_advance()
    key.release(Weapon)

    if Variation ~= "none" then
        movie.frame_advance()
        key.hold(Variation)
        movie.frame_advance()
        key.release(Variation)
    end
end

-- Performs an ssj, takes in the frame ssj you want
function SSJ(frame)
    delay = frame
    key.hold("leftctrl")
    movie.frame_advance()
    key.release("leftctrl")
    movie.frame_advance(delay)
    key.hold("space")
    movie.frame_advance()
    key.release("space")
end
